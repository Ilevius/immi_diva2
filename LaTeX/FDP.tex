\section*{Поле $\boldsymbol{U}_{0}$}%

\[
u_{0}\left(  x,z\right)  =\frac{1}{2\pi}\int\limits_{-\infty}^{+\infty}%
K_{0}\left(  \alpha,z\right)  \boldsymbol{Q}\left(  \alpha\right)  e^{-i\alpha
x}d\alpha
\]
где
\[
K_{0}\left(  \alpha,z\right)  =\left(
\begin{array}
[c]{ll}%
-i\alpha M & -i\alpha P\\
S & R
\end{array}
\right)
\]%
\[
P(\alpha,z)=\frac{1}{\Delta}\left(  -\left(  \alpha^{2}-0,5\varkappa_{2}%
^{2}\right)  e^{\sigma_{1}z}+\sigma_{1}\sigma_{2}e^{\sigma_{2}z}\right)
\]%
\[
R\left(  \alpha,z\right)  =\frac{\sigma_{1}}{\Delta}\left(  -\left(
\alpha^{2}-0,5\varkappa_{2}^{2}\right)  e^{\sigma_{1}z}+\alpha^{2}%
e^{\sigma_{2}z}\right)
\]%
\[
M\left(  \alpha,z\right)  =\frac{-i\sigma_{2}}{\alpha^{2}\Delta}\left(
-\alpha^{2}e^{\sigma_{1}z}+\left(  \alpha^{2}-0.5\varkappa_{2}^{2}\right)
e^{\sigma_{2}z}\right)
\]%
\[
S\left(  \alpha,z\right)  =\frac{-i}{\Delta}\left(  -\sigma_{1}\sigma
_{2}e^{\sigma_{1}z}+\left(  \alpha^{2}-0.5\varkappa_{2}^{2}\right)
e^{\sigma_{2}z}\right)
\]%
\[
\Delta=2\mu\left(  -\left(  \alpha^{2}-0.5\varkappa_{2}^{2}\right)
^{2}+\alpha^{2}\sigma_{1}\sigma_{2}\right)
\]%
\[
\varkappa_{1}^{2}=\frac{\rho\omega^{2}}{\lambda+2\mu},\varkappa_{2}^{2}%
=\frac{\rho\omega^{2}}{\mu},\sigma_{n}=\sqrt{\alpha^{2}-\varkappa_{n}^{2}}%
\]
В точности
как в методичке!

\bigskip\  Пусть $\boldsymbol{Q}(\alpha)=(0;1)$
тогда
\[
\mathbf{U}_{0}(\alpha,z)=\left(
\begin{array}
[c]{l}%
-i\alpha P\\
R
\end{array}
\right)  =\left(
\begin{array}
[c]{l}%
\frac{i\alpha}{\Delta}\left(  \left(  \alpha^{2}-0,5\varkappa_{2}^{2}\right)
e^{\sigma_{1}z}-\sigma_{1}\sigma_{2}e^{\sigma_{2}z}\right) \\
\frac{\sigma_{1}}{\Delta}\left(  -\left(  \alpha^{2}-0,5\varkappa_{2}%
^{2}\right)  e^{\sigma_{1}z}+\alpha^{2}e^{\sigma_{2}z}\right)
\end{array}
\right)  =
\]%
\[
=\left(
\begin{array}
[c]{l}%
i\alpha\left(  \alpha^{2}-0,5\varkappa_{2}^{2}\right)  \frac{1}{\Delta
}e^{\sigma_{1}z}-i\alpha\sigma_{1}\sigma_{2}\frac{1}{\Delta}e^{\sigma_{2}z}\\
-\left(  \alpha^{2}-0,5\varkappa_{2}^{2}\right)  \sigma_{1}\frac{1}{\Delta
}e^{\sigma_{1}z}+\sigma_{1}\alpha^{2}\frac{1}{\Delta}e^{\sigma_{2}z}%
\end{array}
\right)
\]
